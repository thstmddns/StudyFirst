## 4차시

```python
def dfs(total,now):
    global answer
    if total < answer:
        if 0 not in visited[1:]:
            answer = min(answer, total + way[now][0])
            return
        else:
            for j in range(1,n):
                if j != now and visited[j] == 0:
                    visited[j] = 1
                    dfs(total + way[now][j], j)
                    visited[j] = 0


for tc in range(1, int(input())+1):
    n = int(input())
    way = [list(map(int,input().split())) for _ in range(n)]
    answer = 1101

    for i in range(1,n):
        visited = [0] * n
        visited[i] = 1
        dfs(way[0][i], i)

    print(f'#{tc} {answer}')
```

## 사탕 게임

```python
def check(arr):    # 최대 사탕 체크하기
    global maxx, n
    answer = 0

    for i in range(n):
        cnt, cnt2 = 1, 1
        
        for j in range(1,n):
            if arr[i][j-1] == arr[i][j]:
                cnt += 1
            else:
                cnt = 1
            if cnt > answer:
                answer = cnt
            
            if arr[j-1][i] == arr[j][i]:
                cnt2 += 1
            else:
                cnt2 = 1
            if cnt2 > answer:
                answer = cnt2

    maxx = max(maxx, answer)


n = int(input())
case = [list(input()) for _ in range(n)]
case2 = list(map(list,zip(*case)))

maxx = 0
for i in range(n):
    for j in range(n-1):
        if case[i][j] != case[i][j+1]:  # 앞뒤가 다르다면 바꾼상태로 사탕 탐색 시작
            case[i][j], case[i][j+1] = case[i][j+1], case[i][j]
            check(case)
            case[i][j], case[i][j+1] = case[i][j+1], case[i][j] # 탐색이 끝나면 다시 원위치
        
        if case2[i][j] != case2[i][j+1]:  # 앞뒤가 다르다면 바꾼상태로 사탕 탐색 시작
            case2[i][j], case2[i][j+1] = case2[i][j+1], case2[i][j]
            check(case2)
            case2[i][j], case2[i][j+1] = case2[i][j+1], case2[i][j] # 탐색이 끝나면 다시 원위치

print(maxx)
```
