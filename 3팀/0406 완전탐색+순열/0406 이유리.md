def s(M):
    global cnt
    for i in range(10-M+1):
        for j in range(10-M+1):
            if arr[i][j] == 1:
                while 0 < M < 6:
                    for ii in range(i*M):
                        for jj in range(j*M):
                            if arr[ii][jj] == 1:
                               arr[ii][jj] = 0
                                cnt += 1
                    else:
                    	M -= 1
    return -1
arr = [list(map(int, input().split())) for _ in range(10)]
cnt = 0
print(s(5))
