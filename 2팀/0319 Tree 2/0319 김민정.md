### 8차시 이진 힙

1. enq 함수 선언

```python
def enq(item):
    global last
    last += 1
    tree[last] = item
    c = last        # 자식노드 번호
    p = last // 2   # 부모노드 번호
    while p > 0 and tree[p] > tree[c]:  # 최소힙
        tree[p], tree[c] = tree[c], tree[p]
        c = p       # 위로 올라가기
        p = c // 2


T = int(input())
for tc in range(1, T+1):
    N = int(input())
    temp = list(map(int, input().split()))  # 입력 리스트
    tree = [0] * (N+1)
    last = 0    # 힙의 원소의 개수(= 노드번호)
    # 최소 heap 만들기
    for i in temp:
        enq(i)
    # 조상노드 합 구하기
    i = N // 2
    result = 0
    while i >= 1:
        result += tree[i]
        i //= 2
    print(f'#{tc} {result}')
```

2. heapq 라이브러리 활용 (10개 중 3개 맞음)

```python
import heapq

T = int(input())
for tc in range(1, T+1):
    print(f'#{tc}', end=' ')
    N = int(input())
    temp = list(map(int, input().split()))
    heapq.heapify(temp)
    temp = [0] + temp
    i = N // 2
    result = 0
    while i >= 1:
        result += temp[i]
        i //= 2
    print(result)
```

### 9차시 노드의 합

```python
import sys
sys.stdin = open("노드의 합_input.txt")


def postorder(n):   # 후위순회
    if n <= N:
        postorder(n*2)      # 왼쪽 자식노드
        postorder(n*2+1)    # 오른쪽 자식노드
        if n*2+1 <= N:      # 자식노드가 2개인 경우
            tree[n] = tree[n*2] + tree[n*2+1]
        elif n*2 <= N:      # 자식노드가 1개인 경우(완전이진트리니까 왼쪽부터 채워짐)
            tree[n] = tree[n*2]


T = int(input())
for tc in range(1, T+1):
    # N: 노드 개수, M: 리프노드 개수, L: 값 출력할 노드
    N, M, L = map(int, input().split())
    tree = [0] * (N+1)  # 완전 이진트리 => 1차원 리스트
    for i in range(M):
        a, b = map(int, input().split())
        tree[a] = b
    postorder(1)
    print(f'#{tc} {tree[L]}')
```

### 11725 트리의 부모 찾기

* 성공 (인접리스트)

```python
from sys import stdin
input = stdin.readline


def bfs(v):
    global parent
    q = []
    visited = [0] * (N + 1)
    q.append(v)
    visited[v] = 1
    while q:
        v = q.pop(0)
        for node in tree[v]:        # 인접리스트에 있는 노드 번호 확인
            if not visited[node]:   # 방문체크
                q.append(node)
                visited[node] = 1
                parent[node] = v
    return parent


N = int(input())
tree = [[] for _ in range(N+1)]     # 인접리스트
parent = [0]*(N+1)  # 부모노드 정보
for i in range(N-1):
    a, b = map(int, input().split())
    # 방향성 없음
    tree[a].append(b)
    tree[b].append(a)
bfs(1)
for i in range(2, N+1):
    print(parent[i])
```

* 메모리 초과 (인접행렬)

```python
from sys import stdin
input = stdin.readline

N = int(input())
parent = [0] * (N+1)
adj = [[0]*(N+1) for _ in range(N+1)]
stack = [1]
for i in range(N-1):
    a, b = map(int, input().split())
    adj[a][b] = adj[b][a] = 1

while stack:
    p = stack.pop()
    for i in range(N+1):
        if adj[p][i] == 1:
            parent[i] = p
            stack.append(i)
            adj[p][i] = adj[i][p] = 0

for i in range(2, N+1):
    print(parent[i])
```



### 1967 트리의 지름

* 런타임에러

```python
from sys import stdin
input = stdin.readline


def preorder(n):
    global cnt
    if 0 < c2[n] <= N:
        preorder(c1[n])
        preorder(c2[n])
        cnt += max(weight[n][c1[n]], weight[n][c2[n]])
    return


# 완전이진트리가 아님 -> 2차원 배열로 풀이
N = int(input())    # N: 노드의 개수
# adj = [[] for _ in range(N+1)]  # 인접리스트
p = [0] * (N+1)
c1 = [0] * (N+1)
c2 = [0] * (N+1)
weight = [[0] * 101 for _ in range(N+1)]
for i in range(N-1):
    a, b, w = map(int, input().split())
    weight[a][b] = w
    if not c1[a]:
        c1[a] = b
    else:
        c2[a] = b
    p[b] = a

max_value = 0
for i in range(1, N+1):
    cnt = 0
    preorder(i)
    plus = min(weight[i][c1[i]], weight[i][c2[i]])
    if cnt+plus >= max_value:
        max_value = cnt+plus
print(max_value)
```
