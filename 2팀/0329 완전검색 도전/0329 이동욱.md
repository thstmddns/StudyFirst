#### 테트로미노

```python
from sys import stdin

delta = [[1, 0], [-1, 0], [0, 1], [0, -1]]
# 한점을 주어줬을 때 3번이동하면 ㅗ,ㅜ,ㅏ,ㅓ를 제외한 모든 테트리스모형이 된다.
def f(x, y, cnt, v): # x,y는 현재 위치, cnt는 몇번갔는지, v는 value
    global max_value
    # 정지조건
    if cnt == 3:
        print(v)
        if max_value < v:
            max_value = v
            return
    else:
        for a, b in [[1, 0], [-1, 0], [0, 1], [0, -1]]:
            new_x, new_y = x + a, y + b
            if 0 <= new_x < N and 0 <= new_y < M and not visited[new_x][new_y]:
                visited[new_x][new_y] = 1
                f(new_x, new_y, cnt + 1, v + arr[new_x][new_y])



N, M = map(int, stdin.readline().split())
arr = [list(map(int, stdin.readline().split())) for _ in range(N)]
max_value = 0
for i in range(N):
    for j in range(M):
        visited = [[0] * M for _ in range(N)]
        visited[i][j] = 1
        f(i, j, 0, arr[i][j])
        # ㅗ, ㅜ, ㅓ, ㅏ 계산하기
        for a in range(4):
            value = arr[i][j]
            flag = 1
            for b in range(len(delta)):
                if a != b:
                    new_i, new_j = i + delta[b][0], j + delta[b][1]
                    if 0 <= new_i < N and 0 <= new_j < M:
                        value += arr[new_i][new_j]
                    else:
                        flag = 0
                        break
            if flag:
                if value > max_value:
                    max_value = value
print(max_value)
```
