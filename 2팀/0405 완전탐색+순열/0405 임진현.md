```python
def permutation(cnt):
    global maxx
    
    if cnt == 3:   # 4번째 타자는 이미 정해졌으므로 패스!
        cnt += 1
    if cnt == 9:   # 9명의 선수 순서가 정해지면 게임 시작
        total, now = 0, 0
        for g in point:  # 이닝 시작
            a1, a2, a3, out_cnt = 0, 0, 0, 0
            while out_cnt != 3:  # 아웃이 3번 안될때까지 계속 플레이
                temp = g[arr[now]]
                if temp == 0:
                    out_cnt += 1
                elif temp == 1:
                    total += a3
                    a1, a2, a3 = 1, a1, a2
                elif temp == 2:
                    total += a2 + a3
                    a1, a2, a3 = 0, 1, a1
                elif temp == 3:
                    total += a1 + a2 + a3
                    a1, a2, a3 = 0, 0, 1
                else:
                    total += a1 + a2 + a3 + 1
                    a1 = a2 = a3 = 0

                now += 1
                now %= 9
        maxx = max(maxx, total)
        return

    for player in range(1,9):  # 첫번째 선수는 4번타자로 고정
        if not visited[player]:
            visited[player] = 1
            arr[cnt] = player
            permutation(cnt + 1)
            visited[player] = 0


games = int(input()) # 2 ≤ N ≤ 50
point = [list(map(int, input().split())) for _ in range(games)]

arr, visited = [0] * 9, [0] * 9 # 순열 재료
arr[3] = 0    # 4번타자 고정
maxx = 0

permutation(0)
print(maxx)
```
