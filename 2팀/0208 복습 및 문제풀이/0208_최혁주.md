### 1984. 중간 평균값 구하기

```python
#중간 평균값 구하기

a = int(input())

for i in range(a):
    b = list(map(int, input().split()))
    new_b = sorted(b)[1:-1] #최대값과 최소값 제거
    total = 0
    for num in new_b:
        total += num
    
    print(f'#{i+1} {round(total / len(new_b))}')

```



### 1986. 지그재그 숫자

```python
a = int(input()) #테스트 개수

for t in range(a):
    b = int(input())
    lst = []
    for i in range(1,b+1):
        lst.append(i)

    for i in range(1, len(lst)):
        lst[i] = lst[i-1] + (lst[i] * (-1)**i)

    print(f'#{t+1} {lst[-1]}')
```



### 1989. 초심자의 회문검사

```python
#초심자의 회문검사

a = int(input())

lst = []
for i in range(a):
    b = input()
    lst.append(b)

result_lst = []

def palindrome(lst):
    for word in lst:
        if word == word[::-1]:
            result_lst.append(1)
        else:
            result_lst.append(0)
    
    return list(enumerate(result_lst, start = 1))

for i in range(0, len(lst)):
    print(f'#{palindrome(lst)[i][0]} {palindrome(lst)[i][1]}')
```





### 2007. 패턴 마디의 길이

```python
a = int(input()) #테스트 개수

for i in range(a):
    lst = []
    b = input()
    c = list(b) #한글자씩 분해

    for s in range(0,len(c)):
        lst.append(c[s]) #일단 하나씩 넣음
        if c[s+1] == c[0]: #그 다음 글자가 첫글자랑 같을 때(SAMSUNG 케이스)
            if c[s+2] not in lst: #그런데 그 다다음 글자가 아직 리스트에 없을 때 
                continue
            else:
                break
    print(f'#{i+1} {len(lst)}')
```
