#### 파리퇴치(2001)

```python
import sys
sys.stdin = open("input.txt", "r")

# test case 수
T = int(input())

for i in range(1, T+1):
    max_p = 0
    a = []
    N, M = map(int, input().split())
    for j in range(N):
        a.append(list(map(int, input().split())))
    
    for k in range(N-M+1):
        for l in range(N-M+1):
            sum = 0
            for x in range(k, k+M):
                for y in range(l, l+M):
                    sum += a[x][y]
            if sum > max_p:
                max_p = sum
    print(f'#{i} {max_p}')
```

#### 달팽이(1954)

```python
T = int(input())

for tc in range(1, T+1):
    n = int(input())
    # n*n 배열 만들기
    rail = [[0]*n for _ in range(n)]
    # 방향 설정(우, 하, 좌, 상)
    dr = [0, 1, 0, -1]
    dc = [1, 0, -1, 0]
    # 초기 위치, 방향 초기화
    row, column = 0, 0
    move = 0

    # 숫자 1부터 n*n까지
    for j in range(1, n*n+1):
        rail[row][column] = j
        row += dr[move]
        column += dc[move]
        if 0 <= row < n and 0 <= column < n and rail[row][column] == 0:
            pass
        else:
            row -= dr[move]
            column -= dc[move]
            move = (move + 1) % 4
            row += dr[move]
            column += dc[move]

    for k in rail:
        print(*k)
```
