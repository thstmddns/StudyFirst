#### 1. minmax(4828)

```python
import sys
sys.stdin = open('input.txt','r')

def min_max(a): 
    count = 1
    while count != 0: # count 가 0이 되면 반복 종료
        count = 0
        for i in range(len(a)-1): # 순서를 정렬할 때마다 카운트 추가
            if a[i] > a[i+1]:
                a[i], a[i+1] = a[i+1], a[i]
                count +=1
    return a[-1] - a[0]


T = int(input())
for tc in range(1, T+1):
    n = input()
    nums = list(map(int,input().split()))
    print(f'#{tc} {min_max(nums)}')

```

```python
import sys
sys.stdin = open('input.txt','r')


def my_bus(k,n,nums):
#리스트 만들기
    route = [0] * (n+1)
    route[0] = k 
    for i in nums:
        route[i] = k

# 앞으로 가면서 에너지가 하나씩 줌
    energy = route[0]
    count = 0
    for i in range(1, n+1):
        energy -= 1

# 판단
# 1. 남은 연료로 끝까지 갈 수 있는가?
        if i + energy >= n:
            return count

# 2. 충전소를 만났을 때 충전을 안하고 다음 충전소 까지 갈 수 있는가?
        elif route[i] == k:
            if k in route[i+1:i+energy+1]: #다음 충전소 까지 연료가 있다면 진행
                pass
            else : # 다음 충전소까지 연료가 없다면 충전
                count += 1
                energy = k
# 연료가 0 이 되면 0 반환
        elif energy == 0:
            return 0

T = int(input())
for tc in range(1,T+1):
    k, n, m = map(int, input().split())
```

#### 3. 숫자카드(4834)

```python
def my_max_count(nums):
    count_lst = [0]*10
    for i in nums:
        i = int(i)
        count_lst[i] +=1

    # 앞에서 부터 큰 수 확인
    big = 0
    big_index = 0
    for i in range(1,10):
        if big <= count_lst[i]:
            big_index = i
            big = count_lst[i]
    return big_index, big


T = int(input())
for tc in range(1,T+1):
    n = input()
    nums = input()
    print(f'#{tc}', end = ' ')
    print(*my_max_count(nums))            

```

#### 4. 구간합(4835)

```python
import sys
sys.stdin = open('input.txt','r')
def sick(n,m,nums):
    small = 9999999999
    big = 1
    for i in range(n-m+1):
        total = 0
        for j in range(m):
            total += nums[i+j]
        if total > big:
            big = total
        
        if total < small:
            small = total
    
    return big - small

T = int(input())
for tc in range(1,T+1):
    n,m = map(int,input().split())
    nums = list(map(int, input().split()))
    print(f'#{tc} {sick(n,m,nums)}')
```
