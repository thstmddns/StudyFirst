### 18258

```python
from sys import stdin
from collections import deque

N = int(stdin.readline())
que = deque()
for line in stdin:
    cmd = line.split()
    if 'push' in cmd:
        que.append(cmd[1])
    elif 'pop' in cmd:
        print(que.popleft() if que else -1)
    elif 'size' in cmd:
        print(len(que))
    elif 'empty' in cmd:
        print(0 if que else 1)
    elif 'front' in cmd:
        print(que[0] if que else -1)
    elif 'back' in cmd:
        print(que[-1] if que else -1)
```

### 2164

```python
from collections import deque

deq = deque(range(1, int(input()) + 1))
if len(deq) > 1:
    deq.popleft()
while len(deq) > 1:
    deq.append(deq.popleft())
    deq.popleft()
result = deq[0]
print(result)
```

### 11866

```python
N, K = map(int, input().split())
li = list(range(1, N + 1))
ans = []
k = K - 1
idx = 0
while li:
    idx = (idx + k) % len(li)  # 인덱스 넘치지 않게
    ans.append(str(li.pop(idx)))  # 문자로 변환해서 추가
print(f'<{", ".join(ans)}>')
```
