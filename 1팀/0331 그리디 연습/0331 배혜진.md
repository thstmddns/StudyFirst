#### 동전 0

```python
N, K = map(int, input().split())
coin = [int(input()) for _ in range(N)]
cnt = 0
while K:
    c = coin.pop()
    if K >= c:
        cnt += K//c	# 사용된 동전 갯수
        K = K%c		# 남은 돈
print(cnt)
```

#### 회의실 배정

```python
N = int(input())
arr = [tuple(map(int, input().split())) for _ in range(N)]
arr.sort(key=lambda x : (x[1], x[0]))	# 끝나는 시간이 같은 회의 처리
end = arr.pop(0)[1]
cnt = 1
while arr:
    time = arr.pop(0)
    if time[0] >= end:
        end = time[1]
        cnt += 1
print(cnt)
```

#### 행렬

```python
# 1. 바꾸는 함수 실행
def trans(n,m,a):
    global cnt		
    if n<3 or m<3:					# 행렬 크기가 3X3 이하이면 False
        return -1
    for i in range(N-2):			# index error 나기 전까지 
        for j in range(M-2):
            if a[i][j] != B[i][j]:	# A와 B 비교해서 다르면
                cnt += 1			# 연산 횟수 +1
                for k in range(3):	# 3X3 내의 요소들을 바꿔주기
                    for l in range(3):
                        a[i+k][j+l] = 1 - a[i+k][j+l]
                        
# 2. 잘 바뀌었는지 체크하는 함수 실행                        
def check():		
    for i in range(N):
        for j in range(M):
            if A[i][j] != B[i][j]:	# 바꿨는데도 다른 부분 존재 시
                return -1			# -1 반환
    return cnt						# A=B이면 cnt 반환

N, M = map(int, input().split()) 
A = [list(map(int, input().rstrip())) for _ in range(N)]
B = [list(map(int, input().rstrip())) for _ in range(N)]
cnt = 0
trans(N, M, A)
print(check())
```
