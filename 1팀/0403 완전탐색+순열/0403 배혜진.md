#### 부분수열의 합

```python
def check(n, r):		# 조합 구현
    global cnt
    if r==0:
        if sum(T)==S:	# 조합의 합이 S와 같다면 count
            cnt += 1
    elif n < r:
        return
    else:
        T[r-1] = lst[n-1]
        check(n-1, r-1)
        check(n-1, r)

N, S = map(int, input().split())
lst = list(map(int, input().split()))
cnt = 0
for i in range(1, N+1):
    T = [0]*i
    check(N, i)
print(cnt)
```
