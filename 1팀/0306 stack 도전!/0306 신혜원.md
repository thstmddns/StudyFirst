###  좋은 단어

```python
N = int(input())

ans = 0
for _ in range(N):
    words = input()
    stack = []
    for i in range(len(words)):
        if stack and stack[-1] == words[i]:
                stack.pop()
        else:
            stack.append(words[i])
    if not stack:
        ans += 1
print(ans)
```





### 탑(시간 초과)

```python
from sys import stdin

N = int(input())
arr = list(map(int, stdin.readline().split()))
ans = [0]
for i in range(1, N):
    for j in range(i-1, -1, -1):
        if arr[i] < arr[j]:
            print(j+1, end=' ')
            break
    else:
        print(0, end=' ')
        
        
N = int(input())
arr = list(map(int, input().split()))
ans = [0]
for i in range(1, N+1):
    for j in range(i-1, 0, -1):
        if arr[i] < arr[j]:
            ans.append(j)
            break
    else:
        ans.append(0)
print(*ans)
```



## 탑(정답)

```python
N = int(input())
arr = list(enumerate(map(int, input().split()), 1))
stack = [(0, 100000001)]
ans = []
for i in arr:
    if stack[-1][1] > i[1]:
        ans.append(stack[-1][0])
        stack.append(i)
    else:
        while stack[-1][1] < i[1]:
            stack.pop()
        ans.append(stack[-1][0])
        stack.append(i)
print(*ans)
```



## 오큰수(뭐가 틀렸을까요)

```python
N = int(input())
arr = list(map(int, input().split()))
stack = [100000000] # 그냥 큰수
ans = []
for i in range(len(arr)-1, -1, -1): #리스트를 역으로 순회
    if stack[-1] > arr[i]: #stack에 맨 뒤의 수가 배열 보다 클 경우,
        if stack[-1] == 100000000: # stack에 아무것도 없을 때
            ans.append(-1) # -1 추가
            stack.append(arr[i]) # stack에 배열 추가
        else: # stack에 초기 입력 값 제외 수가 있을 때
            ans.append(stack[-1]) # stack의 가장 뒤값 추가
            stack.append(arr[i]) # stack에 배열 추가
    else: #stack에 맨 뒤의 수가 배열 보다 작을 경우,
        while stack[-1] < arr[i]: # 커질 때 까지
            stack.pop() # stack pop
            if stack[-1] == 100000000: #stack[-1]이 큰수와 같으면
                ans.append(-1) # ans에 -1 추가
                stack.append(arr[i]) #stack에 배열 추가
                break 
        else:
            ans.append(stack[-1]) #break 걸리지 않았으면, ans에 stack[-1] 추가
            stack.append(arr[i]) #stack에 배열 추가
print(*ans[::-1]) #거꾸로 입력되어 있어서 역순으로 출력
```
