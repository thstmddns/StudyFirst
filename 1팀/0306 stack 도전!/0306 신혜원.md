###  좋은 단어

```python
N = int(input())

ans = 0
for _ in range(N):
    words = input()
    stack = []
    for i in range(len(words)):
        if stack and stack[-1] == words[i]:
                stack.pop()
        else:
            stack.append(words[i])
    if not stack:
        ans += 1
print(ans)
```





### 탑(시간 초과)

```python
from sys import stdin

N = int(input())
arr = list(map(int, stdin.readline().split()))
ans = [0]
for i in range(1, N):
    for j in range(i-1, -1, -1):
        if arr[i] < arr[j]:
            print(j+1, end=' ')
            break
    else:
        print(0, end=' ')
        
        
N = int(input())
arr = list(map(int, input().split()))
ans = [0]
for i in range(1, N+1):
    for j in range(i-1, 0, -1):
        if arr[i] < arr[j]:
            ans.append(j)
            break
    else:
        ans.append(0)
print(*ans)
```
