```python
from collections import deque

T = int(input())

for test_case in range(1, T+1):
    N, M = map(int, input().split())
    q = deque(enumerate(list(map(int, input().split())), 0))
    ans = q[M][1]
    cnt = 0
    while True:
        idx, t = q.popleft()
        for i in range(len(q)):
            if q[i][1] > t:
                q.append((idx, t))
                break
        else:
            cnt += 1
            if t == ans and idx == M:
                print(cnt)
                break
```

### 1021

```python
from collections import deque

N, M = map(int, input().split())
q = deque(list(range(1, N+1)))
arr = deque(list(map(int, input().split())))


cnt = 0
while q:
    for i in arr:
        if i == q[0]:
            q.popleft()
        elif i != q[0]:
            if q.index(i) < len(q)//2+1:
                while i != q[0]:
                    q.append(q.popleft())
                    cnt += 1
            else:
                while i != q[0]:
                    q.appendleft(q.pop())
                    cnt += 1
            q.popleft()
    else:
        print(cnt)
        break
```

```python
from collections import deque

T = int(input())

for test_case in range(1, T+1):
    cmd = input()
    N = int(input())
    q = input()

    if N == 0:
        q = []
        if 'D' in cmd:
            print('error')
        else:
            print(q)
    elif N == 1:
        q = q[1:-1]
        if cmd.count('D') > 1:
            print('error')
        elif cmd.count('D') == 1:
            print('[]')
        else:
            print(f'[{q[0]}]')
    else:
        q = deque(list(map(int, q[1:-1].split(','))))
        if cmd.count('D') > N:
            print('error')
        else:
            rev = 0
            for i in cmd:
                if i == 'D':
                    if rev%2: # 홀수
                        q.pop()
                    else:
                        q.popleft()
                else:
                    rev += 1
            else:
                if not q:
                    print('[]')
                else:
                    if rev%2:
                        q.reverse()
                    print('[', end='')
                    for i in range(len(q)-1):
                        print(q[i], end=',')
                    print(f'{q[-1]}]')
```
