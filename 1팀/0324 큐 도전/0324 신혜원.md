```python
from collections import deque

T = int(input())

for test_case in range(1, T+1):
    N, M = map(int, input().split())
    q = deque(enumerate(list(map(int, input().split())), 0))
    ans = q[M][1]
    cnt = 0
    while True:
        idx, t = q.popleft()
        for i in range(len(q)):
            if q[i][1] > t:
                q.append((idx, t))
                break
        else:
            cnt += 1
            if t == ans and idx == M:
                print(cnt)
                break
```

### 5430 (시간초과 다시 푸는중)

```python
from collections import deque

T = int(input())

for test_case in range(1, T+1):
    cmd = input()
    N = int(input())
    q = input()

    if N == 0:
        q = []
        if 'D' in cmd:
            print('error')
        else:
            print(q)
    elif N == 1:
        q = q[1:-1]
        if cmd.count('D') > 1:
            print('error')
        elif cmd.count('D') == 1:
            print('[]')
    else:
        q = deque(list(map(int, q[1:-1].split(','))))
        if cmd.count('D') > N:
            print('error')
        else:
            for i in cmd:
                if i == 'D':
                    q.popleft()
                else:
                    q.reverse()
            else:
                if not q:
                    print('[]')
                else:
                    print('[', end='')
                    for i in range(len(q)-1):
                        print(q[i], end=',')
                    print(q[-1], end=']')
                    print()
```
