def bfs(si, sj):
    q = []
    q.append([si, sj])
    visited = [[0] * N for _ in range(N)]
    visited[si][sj] = 1

    while q:
        ci, cj = q.pop(0)

        # 목적지에 도달 할 경우 탐색 종료
        if ci == ei and cj == ej:
            return visited[ci][cj] -1

        # 8방향 탐색
        for di, dj in [[-2, -1], [-2, 1], [2, -1], [2, 1], [-1, -2], [-1, 2], [1, -2], [1, 2]]:
            ni, nj = ci + di, cj + dj
            if 0 <= ni < N and 0 <= nj < N and not visited[ni][nj]:
                q.append([ni,nj])
                visited[ni][nj] = visited[ci][cj] + 1


T = int(input())
for tc in range(1, T+1):
