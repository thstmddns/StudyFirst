## 일곱 난쟁이

```python
def sol():
    x = y = 0
    for i in range(9):
        for j in range(1 + i, 9):
            if total - (arr[i] + arr[j]) == 100:
                x, y = i, j
                return x, y

import sys

arr = []
for i in range(9):
    arr.append(int(sys.stdin.readline()))
arr.sort()
total = sum(arr)

x, y = sol()
for i in range(9):
    if i == x or i == y:
        continue
    else:
        print(arr[i])
```

## 날짜 계산

```python
E, S, M = map(int, input().split())


e, s, m = 1, 1, 1
cnt = 1
while True:
    if E == e and S == s and M == m:
        break
    else:
        e, s, m = (e+1) % 16, (s+1) % 29, (m+1) % 20
        cnt += 1
        if e == 0:
            e = 1
        if s == 0:
            s = 1
        if m == 0:
            m = 1
print(cnt)
```
